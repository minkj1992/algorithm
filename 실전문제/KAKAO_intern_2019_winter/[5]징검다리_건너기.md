def solution(stones, k):
    if len(stones)==1:return stones[0]
    answer = 0
    N = len(stones)
    dp = [0]*N
    for n in range(N):
        if n<k:
            dp[n]=stones[n]
        else:
            dp[n] = min(max(dp[n-k:n]),stones[n])
    answer = max(dp[N-k:N])
    return answer

from functools import lru_cache
def solution(stones, k):
    @lru_cache()
    def cache(n):
        nonlocal k
        if n<k:
            return stones[n]
        else: 
            return min(max([cache(n) for n in range(n-k,n)]),stones[n])

    if len(stones)==1:return stones[0]
    N = len(stones)
    answer = max([cache(n) for n in range(N-k,N)])
    return answer

def solution(stones, k):
    values = sorted(list(set(stones)),reverse=True)
    answer = 0
    
    while values:
        _min = values.pop()
        piv=_min-answer      # piv is adjustedMin
        zeros = 0
        answer = _min
        
        for i in range(len(stones)):
            if stones[i]==0:
                zeros+=1
                if zeros == k:
                    return answer
                continue
            else:
                stones[i]-=piv
                if stones[i]==0:
                    zeros+=1
                    if zeros == k:
                        return answer
                else:
                    zeros = 0

def solution(stones, k):
    def is_possible(mid, k):
        cnt = 0
        for s in stones:
            if s>=mid:
                cnt = 0
                continue
            cnt+=1
            if cnt == k: return False
        return True

    answer = 0
    l, r = 1, max(stones)
    while l<=r:
        mid = (l+r)//2
        if is_possible(mid,k):
            l = mid + 1
            answer = mid
        else:
            r = mid - 1
    return answer
