```python
def solution(n, times):
    times.sort()
    l, r = 1, n * times[len(times) - 1]

    answer = float('inf')
    while l <= r:
        mid = (l + r) // 2
        var = sum([mid // t for t in times])

        if var >= n:
            r = mid-1
            answer = mid
        else:
            l = mid + 1

    return answer
```
