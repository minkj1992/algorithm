```python
# 기회를 주셔서 감사합니다. 열심히 해보겠습니다.
def is_ok(S):
    stack = []
    for symbol in S:
        if symbol == '(':
            stack.append(symbol)
        else:
            if len(stack) == 0:return False
            stack.pop()
    return False if len(stack) != 0 else True

def solution(p):
    if is_ok(p):return p
    for i in range(0,len(p)+1,2):
        if set(list(p[:i]))=={'(',')'}:
            break
    u,v = p[:i],p[i:]
    if is_ok(u):
        return u+solution(v)
    else:
        if len(u)==2:
            return '()'+solution(v)
        else:
            return '('+''.join([')' if v=="(" else "(" for v in u[1:-1]])+')'+ solution(v)
```

```python
def bal_split(w):
    cnt = 0
    for i,v in enumerate(w):
        if v == '(':cnt += 1
        else:cnt -= 1
            
        if cnt == 0:break
    else:
        return w,''
    return w[:i+1],w[i+1:]

def is_correct(u):
    stack = []
    try:
        for s in u:
            if s=='(':stack.append(s)
            else:stack.pop()
        return False if stack else True
    except: return False

def bracket(w):
    if w == '': return ''
    if is_correct(w): return w
    
    u,v = bal_split(w)
    if is_correct(u):
        return u+bracket(v)
    return '('+bracket(v)+')'+''.join([')' if v=='(' else '(' for v in u[1:-1]])
    
def solution(p):
    answer = bracket(p)
    return answer
```
