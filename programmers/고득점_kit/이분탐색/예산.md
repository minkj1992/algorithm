```python
def solution(budgets, M):
    answer = max(budgets)
    if sum(budgets)<=M:return answer

    budgets = sorted(budgets)
    N = len(budgets)
    a = 0; b = answer
    while a<=b:
        mid = (a+b)//2
        for i,bug in enumerate(budgets):
            if bug>mid:break
        tmp_v = sum(budgets[:i])+(N-i)*mid
        if tmp_v == M:return mid
        elif tmp_v > M:
            b = mid-1
        else:
            a = mid+1
            answer = mid
    return answer
```
