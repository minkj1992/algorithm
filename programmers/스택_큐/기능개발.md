```python
import collections
def solution(progresses, speeds):
    tmp = collections.deque([(100-j)//speeds[i] for i,j in enumerate(progresses)])
    top = tmp.popleft()
    answer =[];cnt=1
    while tmp:
        tmp_value = tmp.popleft()
        if top<tmp_value:
            top = tmp_value
            answer.append(cnt)
            cnt =1
        else:cnt+=1
    #마지막값
    answer.append(cnt)
    return answer
```

## 2nd try (`200114`)

```java
import java.util.*;

class Solution {
    public static int divmod(int a, int b) {
        int leftover = a / b;
        if (a % b != 0) {
            leftover++;
        }
        return leftover;
    }


    public int[] solution(int[] progresses, int[] speeds) {
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < progresses.length; i++) {
            int leftover = divmod(100 - progresses[i], speeds[i]);
            queue.offer(leftover);
        }

        int top = queue.poll();
        List<Integer> tmpAnswer = new ArrayList<>();
        int count = 1;

        while (!queue.isEmpty()) {
            int tmpVal = queue.poll();
            if (top < tmpVal) {
                top = tmpVal;
                tmpAnswer.add(count);
                count = 1;
            } else {
                count++;
            }
        }

        int[] answer = new int[tmpAnswer.size()+1];
        int i = 0;
        for (int q: tmpAnswer){
            answer[i]= q;
            i++;
        }
        answer[i] = count;
        return answer;
    }
}
```
