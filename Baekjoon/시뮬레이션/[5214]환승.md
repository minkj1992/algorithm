## 3rd try (0)

- 1등 코드를 나에게 맞게 변형 시켰다.
```python
from collections import deque
MIS = lambda : map(int,input().split())
# N: 역의 수
# K: 하이퍼 튜브가 연결하는 역의 개수
# M: 하이퍼 튜브의 수

def bfs(i,d):
    # station_idx, depth
    s_visited[i] = 1
    queue = deque([(i,d)])
    while queue:
        i,d = queue.popleft()
        if i == N-1:return d
        for h in s2h[i]:
            if h_visited[h]: continue
            h_visited[h] = 1
            for s in h2s[h]:
                if s_visited[s]:continue
                s_visited[s] = 1
                queue.append((s,d+1))
    return -1

N,K,M = MIS()
s2h = [[] for _ in range(N)] # station to hyper
h2s = [[] for _ in range(M)] # hyper to station
for h in range(M):
    for s in MIS():
        s2h[s-1].append(h)
        h2s[h].append(s-1)

s_visited = [0]*N
h_visited = [0]*M
result = bfs(0,1)
print(result)
```
- bfs()
- hyper visited를 통해 백트래킹
- station visited
