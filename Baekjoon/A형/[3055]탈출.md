# 200217
# [탈출](https://www.acmicpc.net/problem/3055)
# . : 비어있는 곳
# * : 물
# X : 돌
# D : 굴
# S : 고슴도치
# out: 고슴도치가 안전하게 비버의 굴로 이동하기 위해 필요한 최소 시간
'''
- 물이 먼저 확장하고, 고슴도치를 이동시킨다. 고슴도치가 죽는다면 queue에 더 넣지 않는다.
- 고슴도치
    - 물 x
    - 돌 x
    - 물이 찰 예정 x
- 물
    - 비어있는 칸과 고슴도치만 가능
    - 돌 x, 굴 x
'''

from collections import deque

def water_expand():
    while w_pos:
        y,x = w_pos.popleft()
        time = water[y][x]
        for ny,nx in (y,x+1),(y,x-1),(y+1,x),(y-1,x):
            if not((0<=ny<R) and (0<=nx<C)): continue
            if water[ny][nx] <= time+1: continue    #@ERROR1 =를 안넣어주어서 메모리 초과가 되었다.
            water[ny][nx] = time+1
            w_pos.append((ny,nx))

def hedge_move(sy,sx,ey,ex):
    time = 0
    water[sy][sx] = -1
    queue = deque([(sy,sx)])
    while queue:
        for _ in range(len(queue)):
            y,x = queue.popleft()
            for ny, nx in (y, x + 1), (y, x - 1), (y + 1, x), (y-1, x):
                if not ((0 <= ny < R) and (0 <= nx < C)): continue
                if (ny,nx) == (ey,ex): return time+1    #굴
                if water[ny][nx] <= time + 1: continue   #돌,물,고슴도치 visited
                water[ny][nx] = -1
                queue.append((ny, nx))
        time+=1
    return -1


R,C = map(int,input().split())
hedgehog = [None,None]
cave = [None,None]
w_pos = deque([])
water = [[float('inf')]*C for _ in range(R)]
board = [input() for _ in range(R)]
for y in range(R):
    for x in range(C):
        if board[y][x] == ".":
            continue
        elif board[y][x] == "*":
            w_pos.append((y,x))
            water[y][x] = 0
        elif board[y][x] == "X":
            water[y][x] = -1
        elif board[y][x] == "D":
            water[y][x] = -1
            cave = [y,x]
        else:
            hedgehog = [y,x]

water_expand()
result = hedge_move(*hedgehog,*cave)
print("KAKTUS")if result == -1 else print(result)
