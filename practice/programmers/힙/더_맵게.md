```py
import heapq as hq

def solution(scoville, K):
    hq.heapify(scoville)
    if scoville[0] >= K:
        return 0
    
    
    answer = 0
    while len(scoville) > 1:
        answer += 1
        mix = hq.heappop(scoville) + hq.heappop(scoville) * 2        
        hq.heappush(scoville, mix)
        
        if scoville[0] >= K:
            return answer
    return -1 
```

```py
import heapq as hq

def solution(scoville, K):
    hq.heapify(scoville)
    answer = 0

        answer += 1
        s1 = hq.heappop(scoville)
        s2 = hq.heappop(scoville)
        
        hq.heappush(scoville, s1 + s2 * 2)
        
        if scoville[0] >= K:
            return answer

    return -1
```
