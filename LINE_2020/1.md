def solution(inputString):
    stack = []
    _open = ['(','{','[','<']
    _close = [')','}',']','>']
    cnt_dict = {'(':0,'{':0,'[':0,'<':0}
    result = 0
    for c in inputString:
        if c in _open:
            cnt_dict[c]+=1
        elif c in _close:
            idx = _open[_close.index(c)]
            cnt_dict[idx]-=1
            result+=1
            if cnt_dict[idx] < 0: return -1

    return result
